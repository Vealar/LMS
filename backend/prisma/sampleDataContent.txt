model Block {
  id          Int          @id @default(autoincrement())
  topicId     Int
  topic       Topic        @relation(fields: [topicId], references: [id])
  type        BlockType
  title       String
  description String?
  deadline    DateTime?
  content     Json?
  order       Int          @default(0)
  attachments Attachment[] // Отношение к вложениям
  createdAt   DateTime     @default(now())
}

model Attachment {
  id        Int      @id @default(autoincrement())
  blockId   Int
  block     Block    @relation(fields: [blockId], references: [id])
  name      String // Имя файла (для отображения)
  url       String // S3 или другая ссылка
  createdAt DateTime @default(now())
}
